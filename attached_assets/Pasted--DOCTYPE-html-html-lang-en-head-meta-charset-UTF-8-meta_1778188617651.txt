<script>

let stumpCounter = 0;

const PRICE_PER_SQ_IN = 0.37;
const BASE_SERVICE_FEE = 100;
const SALES_TAX = 0.07;

function addStump(){

stumpCounter++;

const container =
document.getElementById('stumpContainer');

const div = document.createElement('div');

div.className = 'stump-item';

div.innerHTML = `

<h4>
Stump #${stumpCounter}
</h4>

<div class="grid">

<div>

<label>
Diameter (inches)
</label>

<input
type="number"
class="diameter"
placeholder="Example: 24"
oninput="calculateTotal()"
>

</div>

<div>

<label>
Quantity
</label>

<input
type="number"
class="quantity"
value="1"
min="1"
oninput="calculateTotal()"
>

</div>

</div>

<div class="checkbox">

<input
type="checkbox"
class="deep"
onchange="calculateTotal()"
>

<span>
Extra Depth Grinding (10–12") (+20%)
</span>

</div>

`;

container.appendChild(div);

calculateTotal();

}

function calculateTotal(){

const diameters =
document.querySelectorAll('.diameter');

const quantities =
document.querySelectorAll('.quantity');

const deepOptions =
document.querySelectorAll('.deep');

let stumpAreaTotal = 0;
let stumpCount = 0;

for(let i = 0; i < diameters.length; i++){

const diameter =
parseFloat(diameters[i].value) || 0;

const quantity =
parseInt(quantities[i].value) || 0;

const deep =
deepOptions[i].checked;

const radius = diameter / 2;

const area =
Math.PI * radius * radius;

let stumpPrice =
area * PRICE_PER_SQ_IN;

if(deep){
stumpPrice *= 1.20;
}

stumpAreaTotal += stumpPrice * quantity;

stumpCount += quantity;

}

// BASE FEE + AREA CALCULATION

let baseTotal =
BASE_SERVICE_FEE + stumpAreaTotal;

// DISCOUNTS

let discountRate = 0;

if(stumpCount >= 3 && stumpCount <= 6){
discountRate = 0.10;
}

if(stumpCount >= 7){
discountRate = 0.15;
}

const discountAmount =
stumpAreaTotal * discountRate;

const discountedTotal =
baseTotal - discountAmount;

// SERVICE PACKAGE %

const serviceMultiplier =
parseFloat(
document.getElementById('servicePackage').value
);

const serviceCost =
discountedTotal * serviceMultiplier;

const subtotal =
discountedTotal + serviceCost;

// SALES TAX

const taxAmount =
subtotal * SALES_TAX;

const finalTotal =
subtotal + taxAmount;

// LIVE DROPDOWN PRICE UPDATES

const serviceDropdown =
document.getElementById('servicePackage');

serviceDropdown.options[0].text =
Stump Removal (Leave Wood Chips On Property)';

serviceDropdown.options[1].text =
Stump Removal + Chip & Debris Removal (+$' +
(discountedTotal * 0.75).toFixed(0) + ')';

serviceDropdown.options[2].text =
Removal + Debris Removal + Top Soil & Seed Mat (+$' +
(discountedTotal * 1.05).toFixed(0) + ')';

// UPDATE UI

document.getElementById('totalStumps').innerText =
stumpCount;

document.getElementById('baseCost').innerText =
$' + baseTotal.toFixed(2);

document.getElementById('discount').innerText =
(discountRate * 100) + '%';

document.getElementById('serviceCost').innerText =
$' + serviceCost.toFixed(2);

document.getElementById('salesTax').innerText =
$' + taxAmount.toFixed(2);

document.getElementById('finalTotal').innerText =
$' + finalTotal.toFixed(2);

}

addStump();

</script>
